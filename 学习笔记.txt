1：查询每个类别下id最大的商品
	select goods_id,goods_name,cat_id,shop_price from goods where goods_id in (select max(goods_id) from goods group by cat_id;
	select goods_id,goods_name,cat_id,shop_price from (select goods_id,goods_name,cat_id,shop_price from goods order by cat_id asc,goods_id desc) as tmp group by cat_id;
2:从类别表中取出其类别下有商品的类别(如果该类别下没有商品，则不取出)，[使用where子查询]
	select c.cat_id,c.cat_name from category c where c.cat_id in(select g.cat_id from goods g group by g.cat_id);
	select c.cat_id,c.cat_name from catrgory c where exists(select 1 from goods g where g.cat_id =c.cat_id);
3:any关键字查询
	select cat_id,cat_name from category where cat_id >any(select goods_id form goods);
4:all关键字查询
	select cat_id,cat_name from category where cat_id >all(select goods);
5:左连接查询
	select g.goods_name,g.cat_id,c.cat_id,c.cat_name from goods g LEFT JOIN category c ON g.cat_id = c.cat_id;
	SELECT g.goods_name,g.cat_id,c.cat_id,c.cat_name FROM goods g LEFT JOIN category c ON g.cat_id = c.cat_id;
